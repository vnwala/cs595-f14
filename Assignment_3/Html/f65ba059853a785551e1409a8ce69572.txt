        function submitTagForm(reqType, tagValue) {
            if (reqType.indexOf('add')!=-1) {
                $('#addtags').hide();
                $('#addtags').trigger('hide');
                $('#tag_spinner').show();
            }
            $.ajax({
                 type: 'POST',
                 url: API2_PATH+'editstorytags',
                 data: { story_id: 57064673, request_type: reqType, tag_value : tagValue },
                 success: function(response) {
                     if (reqType.indexOf('add')!=-1) {
                        $('#addprompt').fadeIn();
                        $('#tag_spinner').hide();
                        var title = 'Remove';
                         $.each(response, function(key, value) {
                             $('#tags_inside').append('<span style="margin: 2px" class="b2 small white remove_tag" title="'+title+'">'+ value + ' x' + '</span>');
                         });
                        $('#addtags').val('');
                     }
                 },
                 error: function(response) {
                     var data = $.parseJSON(response.responseText);
                     if (data.message) {
                         alert(data.message);
                     }
                    $('#addprompt').fadeIn();
                    $('#tag_spinner').hide();
                    $('#addtags').val('');
                 }
             });
        }
