function getQueryString(key, default_)
{
  if (default_==null) default_="";
  key = key.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
  var regex = new RegExp("[\\?&]"+key+"=([^&#]*)");
  var qs = regex.exec(window.location.href);
  if(qs == null)
    return default_;
  else
    return qs[1];
}

<script>
//
// $loadMoreRenderer -- the element used to format the new data to be loaded
// $loadMoreCallback -- url used to generate the next page of data
// $loadMoreRotateAds -- if true we will rotate our ads on a load more
//

//
// In startup:
//  * the first page is loaded - LMParameter is set depending on if the last element was supersized
//  * LMNextPage is set to 1, the id of the next page
//  * call the LMRequestNextPage routine
//
// The LMRequestNextPage routine
//  * sets LMReadyPage to false (0) as there is no hidden page ready to be displayed
//  * sets LMScrollRequestMade to false as the user has not yet asked for the page we are loading
//  * makes an asynchronious request for a page LMNextPage to be rendered into a hidden div of id LMNextPage
//					its passed LMNextPage and LMLastWasSupersized
//
// The render page request:
//  * renders the page into a hieed div with id LMPageNext
//  * calls an update routine LMPageRendered, passing it a boolean indicating if the last element rendered was supersized
//
// LMPageRendered sets:
//    - LMReadyPage is set to LMNextPage
//    - LMNextPage++
//    - if LMScrollRequestMade then the user has requested that a page be made visible, so call LMDisplayPage()
//
//  If the user presses the 'load more' button:
//  * if there is no page ready (LMReadyPage == false) we:
//    - remeber that the user pressed the 'load more' button so set LMScrollRequestMade = true
//
//  * if there is a page ready (LMReadyPage != false) we:
//    - make the hidden div with id LMReadyPage visible
//    - LMReadyPage = 0 (no longer ready)
//    - make a request for the next page by calling LMRequestNextPage
//

LMUrlData = 'speed_reads';
LMReadyPage = 0;						// 0 if no page is ready to be displayed, N, where N is the id of a hidden div is a div is ready
LMNextPage = 0;							// the next page to display (it may be being loaded or be hidden)
LMParameter = 0;						// was the last element on the last page we received from the server supersized?
LMScrollRequestMade = 0;				// has the user hit "load more' and we have not done so?
LMEnableConsole = false;
LMEndArray = [];

// Set up a asynchronious request to load page 1
function LMStartup() {
	if (LMEnableConsole) {console.log('LMStartup');};
    LMNextPage = 1;			// we use php to render page 0 directly.  We will load page 1 asynchronously
	LMRequestNextPage();				// ask for the page 1 to be sent
}

function LMRequestNextPage() {
	if (LMEnableConsole) {console.log('LMRequestNextPage LMNextPage:'+LMNextPage+' LMParameter:'+LMParameter+
								' LMUrlData: '+LMUrlData);};
    LMReadyPage = 0;					// The page we are requesting is not ready - duh
	LMScrollRequestMade = 0;			// The user cannot have request that we display it yet - we only just showed him the last page
	new Ajax.Updater('LoadMoreInsertionPoint',
						'/section/speedReadsLoadMore/'+LMNextPage+'/'+LMParameter+'/'+LMUrlData,
						{
							method: 'get',
							asynchronous:true,
							evalScripts:true,
							insertion:'before',
							onComplete: LMPageRendered,
							requestHeaders:['X-Update', 'LoadMoreInsertionPoint']});
	if (LMEnableConsole) {console.log('LMRequestNextPage EXIT LMReadyPage:'+LMReadyPage+' LMScrollRequestMade:'+LMScrollRequestMade);};
	if (true) {rotateAds();};
}

function LMPageRendered() {
	if (LMEnableConsole) {console.log('LMPageRendered LMNextPage:'+LMNextPage+' LMScrollRequestMade:'+LMScrollRequestMade);};
    // get supersized value from the new div LMSSNextPage
	LMParameter = jQuery("#LMSS"+LMNextPage)
					.text();	// remember if the last element was supersized
	if (LMEnableConsole) {console.log('LMParameter:'+LMParameter);};
    LMReadyPage = LMNextPage;				// we have a new page that can be displayed and it's id is LMNextPage
	LMNextPage++;
	if (LMScrollRequestMade) {
	    LMLoadMore();
	}
	if (LMEnableConsole) {console.log('LMPageRendered EXIT LMReadyPage:'+LMReadyPage);};
}

// display the hidden page
function LMDisplayPage() {

       jQuery('#LoadMoreTab').hide();

<h2 class="entryHeader">
                <a href="/speedreads/index/268722/speedreads-wind-turbines-are-killing-more-bats-than-ever"
                     onSubmit="rolwa(this,'Home','Home-scroll','position:0.2',''); ">
                      Wind turbines are killing more bats than ever                </a>
        </h2>
        </div>
                <div class="topicEntry" >

                        
                        
                                <a href="/speedreads/index/268722/speedreads-wind-turbines-are-killing-more-bats-than-ever"  >

                                <div class="topicThumb">
                                                                <img src="https://7e8c.https.cdn.softlayer.net/807E8C/origin.theweek.com/img/speedreads/268722.jpg?209.jpg"
                                                                                        alt="Andrew Caballero-Reynolds/Getty Images" width="150" style="margin-right:5px;">
                                                                                  <div class="imgcap">Andrew Caballero-Reynolds/Getty Images</div>
                </div>
        </a>
        

		
			      		 	<p>A new study has found that wind turbines are causing "unprecedented numbers" of fatalities in tree bats.</p>
<p>The study, conducted by researchers at the U.S. Geological Survey's Fort Collins Center in Colorado and <a href="http://www.pnas.org/content/early/2014/09/24/1406672111" target="_blank">published in <em>PNAS</em></a>, found that tree bats are more susceptible to wind turbine deaths, because the turbines' wind current resembles the wind currents surrounding tall trees. The bats <a href="https://www.youtube.com/watch?v=mttpX53OCwU&amp;list=UUeXH8GZyV3sVqAr45AvupOA" target="_blank">approach the turbines</a> expecting to find the resources they'd find in trees, such as a place to rest or insects to eat &mdash; and end up dying instead.</p>
<p>"These bats have been around for millions of years, but nothing in their history would prepare them to recognize wind turbines," Paul Cryan, a research biologist at the U.S. Geological Survey and author of the study, told <em>The Week</em>.</p>
<p>So, what can we do to reduce the bat fatalities at the turbines? The wind turbines already have flashing red lights, which can help bats distinguish the turbines from trees, but Cryan's study also found that changes in the turbines' operation could reduce bat deaths, too. Most of the bat fatalities the researchers observed occurred when average windspeeds were lower, though more research is needed to determine why this is. Cryan said that fatalities could be reduced by waiting until the wind speed is higher to start the turbines' rotation, so the bats will be able to sense the turbulence and avoid the turbines.</p>
<p>Since adult bats reproduce very slowly, their deaths could have lasting effects on the species' population. "We know that thousands of bats are being found beneath wind turbines each year," Cryan told <em>The Week</em>. "These populations may be doing okay, they may be headed toward extinction, we simply don't know based on the information we have right now."</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span><a href="/author/meghan-demaria" ><span class="byline">Meghan&nbsp;DeMaria</span></a></span>

                      45 dogs dead in North Las Vegas fire                </a>
        </h2>
        </div>
                <div class="topicEntry" >

                        
                        
                                <a href="/speedreads/index/268983/speedreads-45-dogs-dead-in-north-las-vegas-fire"  >

                                <div class="topicThumb">
                                                                <img src="https://7e8c.https.cdn.softlayer.net/807E8C/origin.theweek.com/img/speedreads/268983.jpg?209.jpg"

			      		 	<p>Authorities announced Monday that at least 45 dogs died in a house fire in Las Vegas over the weekend.</p>
<p>The fire was reported to the North Las Vegas Fire Department on Sunday night. When the fire department arrived at the home, in a suburb of Las Vegas, they found dogs "running in the front and backyards, but most were found inside" the three-bedroom home, <a href="http://bigstory.ap.org/article/e439bbd26fb24496a65988744c15a604/41-dogs-dead-north-las-vegas-house-fire" target="_blank"><em>The Associated Press</em> reports</a>.</p>
<p>While 45 dogs were reported dead, another 48 survived the fire, <em>AP</em> notes. Some of the dogs were revived with pet-rescue oxygen masks. In addition to the dogs, a man who was at the home <a href="http://www.ksby.com/news/41-dogs-dead-in-north-las-vegas-house-fire/" target="_blank">was taken</a> to University Medical Center in Las Vegas, where he was in stable condition as of Monday morning. Meanwhile, animal-control officers removed the dogs from the house. It's not yet known why so many animals were living there.</p>
<p>Isaac Barron, a local councilman and high school teacher, who owned the house, <a href="http://news.yahoo.com/41-dogs-dead-north-las-vegas-house-fire-180547724.html" target="_blank">issued a statement</a> Monday saying he was heartbroken about the animals' deaths. Barron added that he had been in the process of evicting his tenants, two men who have not yet been identified, from the home when the fire occurred. Police Sgt. Chrissie Coon told <em>AP</em> that the tenants are under investigation for their treatment of the animals. "As an animal lover, I am heartbroken about this tragedy," Barron <a href="http://www.fox5vegas.com/story/26652710/fd-41-dogs-dead-after-north-las-vegas-house-fire" target="_blank">said in the statement</a>. "I hope those responsible are held accountable by the authorities."</p>
<p>North Las Vegas Fire Department Capt. Cedric Williams told <em>AP</em> that police will be "continuing their investigation" of the incident, which he said appeared to be an accident that resulted from an electrical issue. The home, which dated to 1954, was a total loss after the smoke damage, Williams told <em>AP</em>.</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span><a href="/author/meghan-demaria" ><span class="byline">Meghan&nbsp;DeMaria</span></a></span>

                      Man could go to jail for kicking stray cat                </a>
        </h2>
        </div>
                <div class="topicEntry" >

                        
                        

		
			      		 	<p>Andre Robinson was arrested after a video of him kicking a stray cat several yards was posted on Facebook. He now faces jail time. According to the <em><a href="http://www.nytimes.com/2014/09/30/nyregion/animal-abuse-gains-traction-as-a-serious-crime-with-jail-more-often-the-result.html?hp&amp;action=click&amp;pgtype=Homepage&amp;version=HpSumSmallMediaHigh&amp;module=second-column-region&amp;region=top-news&amp;WT.nav=top-news&amp;_r=0" target="_blank">The New York Times</a></em>, however, "had it been a person he kicked, Robinson most likely would have received a quick plea bargain requiring no jail time" &mdash; if he had been arrested at all.</p>
<p>"With one seemingly unconsidered kick," the <em>Times </em>reports, Robinson has "unwittingly placed himself at the center of an impassioned, growing debate" between powerful animal rights activists and lawyers defending the accused who "say that punishment in such cases can seem disproportionate to the crime when an animal is the victim."</p>
<p>Animal rights activists show up every time Robinson appears in court in Brooklyn, surrounding him and calling for a jail sentence. The FBI has begun categorizing animal abuse as a separate crime, and has begun taking such cases more seriously.</p>
<p>Robinson has yet to receive a plea offer from prosecutors, which the <em>Times</em> notes is "extremely rare in misdemeanor cases." Robinson faces up to a year of jail time if convicted. Read the full story <a href="http://www.nytimes.com/2014/09/30/nyregion/animal-abuse-gains-traction-as-a-serious-crime-with-jail-more-often-the-result.html" target="_blank">at <em>The New York Times</em>.</a></p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span>Teresa&nbsp;Mull</span>

			      		 	<p>Vice President Joe Biden announced Monday that the United States will give $450 million in grants to community colleges that are working with employers on job training.</p>
<p>The U.S. will award the grants to hundreds of community colleges across the nation, with the largest grants &mdash; roughly $20 million each &mdash; going to Massasoit Community College in Massachusetts and Chippewa Valley Technical College in Wisconsin. <a href="http://bigstory.ap.org/article/c1d9495e11a6442fbda59f9ce7201641/us-awards-450m-link-job-training-industry" target="_blank"><em>The Associated Press</em> reports</a> that the jobs initiative underscores the Obama administration's plans to improve wages and lower unemployment rates by "linking training to industry demand," with the hopes of growing the American economy.</p>
<p>A significant portion of the grants will be given to training in the cybersecurity and IT-related fields, which the White House noted are in need of employees. One program in Maryland, for example, will include partnerships with Raytheon and IBM to train workers for jobs in information technology or cybersecurity. Other fields the White House said are <a href="http://www.whitehouse.gov/live/vice-president-biden-delivers-remarks-event-announcing-winners-job-driven-training-grants" target="_blank">in need of employees</a> include healthcare, energy, and advanced manufacturing.</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span><a href="/author/meghan-demaria" ><span class="byline">Meghan&nbsp;DeMaria</span></a></span>

                      GOP Senate candidate Ernst pulling ahead in Iowa                </a>
        </h2>
        </div>
                <div class="topicEntry" >

                        
                        
                                <a href="/speedreads/index/268979/speedreads-gop-senate-candidate-ernst-pulling-ahead-in-iowa"  >

                                <div class="topicThumb">
                                                                <img src="https://7e8c.https.cdn.softlayer.net/807E8C/origin.theweek.com/img/speedreads/268979.jpg?209.jpg"
                                                                                        alt="Facebook/JoniforIowa" width="150" style="margin-right:5px;">

			      		 	<p>A new poll published by the <em><a href="http://www.desmoinesregister.com/story/news/elections/2014/09/27/iowa-poll-joni-ernst-leads-bruce-braley/16351013/" target="_blank">Des Moines Register</a></em> puts Iowa State Senator Joni Ernst (R) six points ahead of Democratic challenger Bruce Braley (44-38 percent), while 12 percent of likely voters remain undecided.</p>
<p>Braley has been heavily favored to win the race until just recently, and Ernst has made considerable gains since she won her party's nomination in June.</p>
<p>"Just seven months ago," the <em>Register</em> reports, "political analysts considered Braley almost a shoo-in for a seat held for 30 years by liberal Democrat Tom Harkin."</p>
<p>Ernst and Braley participated in a <a href="http://www.washingtontimes.com/news/2014/sep/28/joni-ernst-bruce-braley-spar-over-immigration-in-f/" target="_blank">heated debate</a> (the first on their campaign trail) in which immigration took center stage. Braley blamed House Republicans for failing to reform immigration policy, while Ernst defended the GOP's anti-amnesty policy.</p>
<p>Braley is a lawyer who has been a member of Congress for eight years. His polling numbers are down in his home district of northeast Iowa, where voters take issue with his negligent behavior concerning veterans' affairs and his role in crafting ObamaCare.</p>
<p>Ernst markets herself as a "mother and soldier" and proud native Iowan. The consensus of political analysts is that Ernst did well in the debate against Braley, who "<a href="http://www.bloomberg.com/news/2014-09-29/braley-fails-to-do-damage-to-ernst-in-iowa-senate-debate.html" target="_blank">failed to do her damage</a>." Democrats hope to use Ernst's stance against raising the minimum wage to their advantage.</p>
<p>The Iowa race is still very close. The <em><a href="http://www.huffingtonpost.com/2014/09/29/gop-iowa-poll_n_5899544.html" target="_blank">Huffington Post</a></em> reports that models give Ernst a "52 percent chance of winning in November."</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span>Teresa&nbsp;Mull</span>

<p class="p1">In December 2012, Belcher shot and killed his girlfriend, Kasandra Perkins. He then drove to the Chiefs' practice facility and, in front of team officials, shot himself in the parking lot.</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span><a href="/author/jon-terbush" ><span class="byline">Jon&nbsp;Terbush</span></a></span>

        </h2>

                                <a href="/speedreads/index/268975/speedreads-spains-prime-minister-will-ask-for-a-court-ban-on-catalonias-independence-vote"  >

                                <div class="topicThumb">

			      		 	<p>Spain's Prime Minister Mariano Rajoy announced Monday that he plans to ask the country's Constitutional Court to ban a new law that would allow the semi-autonomous Catalonia region to vote on its independence.</p>
<p>Rajoy <a href="https://www.thespainreport.com/11504/rajoy-spanish-government-to-ask-constitutional-court-to-strike-down-new-catalan-referendum-law/" target="_blank">said the new Catalan law</a> is "anti-democratic" and that the vote is "not compatible with the Spanish constitution." <a href="http://www.bbc.com/news/world-europe-29410493" target="_blank">His statement</a> comes after Catalonia's President Artur Mas <a href="http://www.bbc.com/news/world-europe-29390774" target="_blank">signed a decree on Saturday</a> that called for a referendum on independence, which would be held on Nov. 9. The statement and referendum also come after Scotland's rejection of independence from the U.K.</p>
<p>Mas, meanwhile, is determined for Catalonia's pro-independence movement to succeed. "Catalonia wants to speak," he <a href="http://www.bbc.com/news/world-europe-29390774" target="_blank">said at the referendum signing</a> on Saturday. "Wants to be heard. Wants to vote. Now is the right time, and we have the right legal framework to do so." Mas says that local laws allow for the regional independence vote, because it would be "non-binding," <a href="http://time.com/3444788/spain-catalonia-independence/" target="_blank"><em>Time</em> reports</a>.</p>
<p>But Rajoy maintains that Spain's constitution doesn't allow referendums on sovereignty that don't include all Spanish citizens. "There is no one and nothing that can deprive Spaniards of their constitutional rights," <a href="https://www.thespainreport.com/11504/rajoy-spanish-government-to-ask-constitutional-court-to-strike-down-new-catalan-referendum-law/" target="_blank">Rajoy said</a>.</p><span class="byline"> -<span style="font-size:1px;">&nbsp;</span>-<span style="font-size:1px;">&nbsp;</span><a href="/author/meghan-demaria" ><span class="byline">Meghan&nbsp;DeMaria</span></a></span>
