<script type="text/javascript">
$(window).load(function(){


 var defaultSize = [505, 400]; // expanded box size: width , height

    $('#grid').masonry({
        singleMode: false,
        columnWidth: 255,
        resizeable: true,
        itemSelector: '.box, .box-nonex, .box-nonex-half',
        animate: true
    });
    
    $('.box').click(function(){
        if ($(this).is('.expanded')){
            restoreBoxes();
        } else {
     var size = ($(this).attr('data-size')) ? $(this).attr('data-size').split(',') : defaultSize;
            $(this)
                // save original box size
                .data('size', [ $(this).width(), $(this).height() ])
                .animate({
                    width: size[0],
                    height: size[1]
                }, function(){
                    // show hidden content when box has expanded completely
                    $(this).find('.expandable').show('slow');
                    $(this).find('.hideable').hide('slow');
                    $('#grid').masonry();
                });
            restoreBoxes();
            $(this).addClass('expanded');
        }
        
        function restoreBoxes(){
            var len = $('.expanded').length - 1;
            $('.expanded').each(function(i){
                var box = $(this).data('size');
                $(this).find('.expandable').hide('slow');
                $(this).find('.hideable').show('slow');
                $(this).animate({
                        width: ( box[0] || 100 ),
                        height: ( box[1] || 'auto' )
                    }, function(){
                        if (i >= len) {
                            $('#grid').masonry();
                        }
                    })
                    .removeClass('expanded');
            })
         }
    });
});
</script>
